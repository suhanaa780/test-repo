class SellingByCatgz:
    def __init__(self):
        self.cost_price = 0
        self.profit_percentage = 0
        self.selling_price = 0
        self.category = ''
    
    def input_cost_price(self):
        self.cost_price = int(input('Enter the cost price: \t '))
        
    def input_category(self):
        self.category = input('Enter the Category: \t ')

    def calc_profit_percentage(self):
        if self.category == 'A' or self.category == 'a':
            self.profit_percentage = 8.5/100
        elif self.category == 'B' or self.category == 'b':
            self.profit_percentage = 0.105
        elif self.category == 'C' or self.category == 'c':
            self.profit_percentage = 12.5/100
        else:
            print ('Invalid Category Entered')
            self.category = 'Invalid'
#            break
        
    def calc_selling_price(self):
        self.selling_price = self.cost_price + (self.cost_price * self.profit_percentage)
        
    def display(self):
        # calling the methods calc of selling price
        self.calc_profit_percentage()
        if not self.category == 'Invalid':            
            self.calc_selling_price()
            print ('Selling Price : \t ',self.selling_price )
